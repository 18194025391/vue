    <path id="upArrow" d="M0 105 L10 110 L0 115" 
            stroke="#409EFF" stroke-width="2" fill="#FFF"
            style="display:none; user-select: none"/>
    <text x="15" y="115" style="font-size: 16px; fill: #409EFF; font-weight: bold; user-select: none">
        info
    </text>
</g>
<script>
    const change = () => {
        document.querySelectorAll('text[id^="info-"]').forEach(el => {
            el.style.display = isHidden ? 'none' : 'inline';
        });
        svg.setAttribute('height', isHidden ? '120px' : maxHeight);
        document.getElementById('upArrow').style.display = isHidden ? 'inline' : 'none';
        document.getElementById('downArrow').style.display = isHidden ? 'none' : 'inline';
    }

    let isHidden = false;
    let svg = document.querySelector('svg');
    let maxHeight = svg.getAttribute('height');
    if(document.getElementById('hidden')){
        isHidden = true;
    }
    change();
    document.getElementById('hideButton').addEventListener('click', function() {
        isHidden = !isHidden;  
        change();
    });
</script>
